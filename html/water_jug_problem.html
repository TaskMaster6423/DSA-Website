struct State { int a, b; };

    queue> q;
    q.push({{0, 0}});

    while(!q.empty()) {
        vector path = q.front(); q.pop();
        State curr = path.back();

        if (curr.a == target || curr.b == target) {
            printPath(path);
            return;
        }

        // Try 6 operations:
        // 1. Fill A, 2. Fill B
        // 3. Empty A, 4. Empty B
        // 5. Pour A->B, 6. Pour B->A
        // If next state !visited, push to queue
    }
}
