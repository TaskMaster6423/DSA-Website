void floydWarshall(int dist[][V], int V) {
    // k is the intermediate node
    for (int k = 0; k < V; k++) {
        // i is source
        for (int i = 0; i < V; i++) {
            // j is destination
            for (int j = 0; j < V; j++) {
                // If vertex k is on the shortest path from i to j,
                // then update the value of dist[i][j]
                if (dist[i][k] + dist[k][j] < dist[i][j])
                    dist[i][j] = dist[i][k] + dist[k][j];
            }
        }
    }
}
