struct Edge { int to; int weight; };

int heuristic(Node* a, Node* b) {
    return sqrt(pow(a->x - b->x, 2) + pow(a->y - b->y, 2));
}

void aStar(int startNode, int endNode) {
    priority_queue> openSet;
    gScore[startNode] = 0;
    fScore[startNode] = heuristic(startNode, endNode);
    openSet.push({-fScore[startNode], startNode});

    while(!openSet.empty()) {
        int u = openSet.top().second; openSet.pop();
        if (u == endNode) return reconstructPath();

        for (Edge& e : adj[u]) {
            int tentativeG = gScore[u] + e.weight;
            if (tentativeG < gScore[e.to]) {
                parent[e.to] = u;
                gScore[e.to] = tentativeG;
                fScore[e.to] = gScore[e.to] + heuristic(e.to, endNode);
                openSet.push({-fScore[e.to], e.to});
            }
        }
    }
}
