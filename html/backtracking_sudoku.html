bool solveSudoku(int grid[N][N]) {
    int row, col;
    if (!findUnassigned(grid, row, col))
        return true; // Success!

    for (int num = 1; num <= 9; num++) {
        if (isSafe(grid, row, col, num)) {
            grid[row][col] = num;
            
            if (solveSudoku(grid))
                return true;

            grid[row][col] = 0; // Backtrack
        }
    }
    return false;
}
