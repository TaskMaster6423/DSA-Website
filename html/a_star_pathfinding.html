// Heuristic: Manhattan Distance
int h(Node a, Node b) {
    return abs(a.x - b.x) + abs(a.y - b.y);
}

void aStar(Node start, Node end) {
    openSet.push(start);
    start.g = 0;
    start.f = h(start, end);

    while (!openSet.empty()) {
        Node current = openSet.popLowestF();
        if (current == end) return reconstructPath(current);

        closedSet.add(current);

        for (Node neighbor : current.neighbors) {
            if (closedSet.contains(neighbor) || neighbor.isWall) continue;

            int tempG = current.g + 1;
            if (tempG < neighbor.g) {
                neighbor.parent = current;
                neighbor.g = tempG;
                neighbor.f = neighbor.g + h(neighbor, end);
                if (!openSet.contains(neighbor)) openSet.push(neighbor);
            }
        }
    }
}
